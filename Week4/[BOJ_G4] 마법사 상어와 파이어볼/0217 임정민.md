```
1. 모든 파이어볼 이동
2. 여러개의 파이어볼이 동일한 좌표에 있을시
yes - 파이어볼 분열
no - 끝

근데 K를 안 사용하고 N을 사용해서 틀린 부분을 1시간동안 찾은 그런 문제
```
```c++
#include <iostream>
#include <vector>
#define MAX 55
using namespace std;

int dx[]={-1,-1,0,1,1,1,0,-1};
int dy[]={0,1,1,1,0,-1,-1,-1};
int O_dir[]={0,2,4,6};
int T_dir[]={1,3,5,7};
struct FIREBALL{
    int x;
    int y;
    int massive;
    int speed;
    int dir;
}; // 파이어볼의 정보를 담아 자료형으로 관리

vector<FIREBALL> a[MAX][MAX];
vector<FIREBALL> fireball;
int N,M,K;
void Move_fireball(){
    for(int i=1;i<=N;i++){
        for(int j=1;j<=N;j++){
            a[i][j].clear();
        }
    }
    for(int i=0;i<fireball.size();i++){
        int x = fireball[i].x;
        int y = fireball[i].y;
        int mass = fireball[i].massive;
        int speed = fireball[i].speed;
        int dir = fireball[i].dir;

        int move = speed%N;
        int nx = x+dx[dir]*move;
        int ny = y+dy[dir]*move;
        if(nx>N)nx-=N;
        if(ny>N)ny-=N;
        if(nx<1)nx+=N;
        if(ny<1)ny+=N;
        a[nx][ny].push_back({nx,ny,mass,speed,dir});
        fireball[i].x = nx;
        fireball[i].y = ny;
    }
}
void Separate_fireball(){
    vector<FIREBALL> temp;
    for(int i=1;i<=N;i++){
        for(int j=1;j<=N;j++){
            if(a[i][j].size()==0)continue;
            if(a[i][j].size()==1){
                temp.push_back(a[i][j][0]);
                continue;
            }
            int massive_sum = 0;
            int speed_sum =0;
            int cnt = a[i][j].size();

            bool Even = true;
            bool Odd= true;
            for(int k=0;k<a[i][j].size();k++){
                massive_sum += a[i][j][k].massive;
                speed_sum += a[i][j][k].speed;
                if(a[i][j][k].dir%2==0)Odd=false;
                else Even = false;
            }
            int mass = massive_sum/5;
            int speed = speed_sum/cnt;
            if(mass==0)continue;
            if(Even==true|| Odd==true){
                for(int k=0;k<4;k++){
                    temp.push_back({i,j,mass,speed,O_dir[k]});
                } 

            }else{
                for(int k=0;k<4;k++){
                    temp.push_back({i,j,mass,speed,T_dir[k]});
                }
            }
        }

    }
    fireball=temp;
}
int main(){
    cin >> N >> M >> K;
    for(int i=0;i<M;i++){
        int r,c,m,s,d;
        cin >> r >> c >> m >> s >> d;
        fireball.push_back({r,c,m,s,d});
        a[r][c].push_back({r,c,m,s,d});
    }

    for(int i=0;i<K;i++){
        Move_fireball();
        Separate_fireball();
    }
    int ans=0;
    for(int i=0;i<fireball.size();i++){
        ans+=fireball[i].m;
    }
    cout << ans<<"\n";
    return 0;
}
