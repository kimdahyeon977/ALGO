```python
import sys
from pprint import pprint
input = sys.stdin.readline

n, m = map(int, input().split())
arr = []
for _ in range(n):
    arr.append(list(map(int, input().split())))
    
moves = [list(map(int, input().split())) for _ in range(m)]

dir_x_list = [-1,-1,0,1,1,1,0,-1]
dir_y_list = [0,-1,-1,-1,0,1,1,1]

check_x_list = [-1,1,1,-1]
check_y_list = [-1,-1,1,1]

cloud = [[0,n-1],[1,n-1],[0,n-2],[1,n-2]]

for d,t in moves:
    # 1. 구름 이동
    cloud = [[(c[0] + dir_x_list[d-1]*t) % n, (c[1] + dir_y_list[d-1]*t) % n] for c in cloud]
    #2. 구름에 비내리기 & 구름 삭제
    for x,y in cloud:
        arr[y][x] += 1
    #pprint(arr)
    #3. 대각선 방향에 물이 있는지 확인
    for c in cloud:
        cnt = 0
        for d in range(4):
            dx = (c[0] + check_x_list[d])
            dy = (c[1] + check_y_list[d])
            if 0 <= dx < n and 0 <= dy < n and arr[dy][dx] > 0:
                cnt += 1
        arr[c[1]][c[0]] += cnt
    #pprint(arr)
    ori_cloud = cloud
    cloud = []
    #4. 구름 생성
    for c in range(n):
        for r in range(n):
            if arr[c][r] >= 2 and [r,c] not in ori_cloud:
                cloud.append([r,c])
                arr[c][r] -= 2
answer = 0
for c in arr:
    for r in c:
        answer += r
print(answer)
```

```python
import sys
input = sys.stdin.readline

n, m = map(int, input().split())
arr = []
for _ in range(n):
    arr.append(list(map(int, input().split())))
    
moves = [list(map(int, input().split())) for _ in range(m)]

dir_x_list = [-1,-1,0,1,1,1,0,-1]
dir_y_list = [0,-1,-1,-1,0,1,1,1]

check_x_list = [-1,1,1,-1]
check_y_list = [-1,-1,1,1]

cloud = [[0,n-1],[1,n-1],[0,n-2],[1,n-2]]

for d,t in moves:
    visited = [[0]*n for _ in range(n)]
    # 1. 구름 이동
    for c in cloud:
        cx = (c[0] + dir_x_list[d-1]*t) % n
        cy = (c[1] + dir_y_list[d-1]*t) % n
    #2. 구름에 비내리기 & 구름 삭제
        arr[cy][cx] += 1
        c[0], c[1] = cx, cy
        visited[cy][cx] = 1
    #3. 대각선 방향에 물이 있는지 확인
    for c in cloud:
        cnt = 0
        for d in range(4):
            dx = (c[0] + check_x_list[d])
            dy = (c[1] + check_y_list[d])
            if 0 <= dx < n and 0 <= dy < n and arr[dy][dx] > 0:
                arr[c[1]][c[0]] += 1

    tmp_cloud = []
    #4. 구름 생성
    for c in range(n):
        for r in range(n):
            if arr[c][r] >= 2 and visited[c][r] == 0:
                tmp_cloud.append([r,c])
                arr[c][r] -= 2
    cloud = tmp_cloud

answer = 0
for c in arr:
    for r in c:
        answer += r
print(answer)
```
