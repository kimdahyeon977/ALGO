'''
틀린 풀이 : 백트래킹
Recusion error, 왜 틀린지 모르겠음
'''
import sys

input = sys.stdin.readline

n,k = map(int, input().split())
lst = list(map(int, input().split()))
max_val = 0

def back(n, cnt, idx):
    global max_val
    if n >= k:
        if n == k:
            max_val = max(max_val, cnt)
        return
    
    if idx >= len(lst):
        return
    
    if lst[idx] % 2 != 0:
        back(n+1, cnt, idx + 1)
        back(n, 0, idx + 1)
    else:
        back(n, cnt + 1, idx + 1)

for num, i in enumerate(lst):
    if i % 2 == 0:
        back(0, 1, num)

print(max_val)
'''
맞은 풀이 : 투포인터
'''
import sys
input = sys.stdin.readline

n, k = map(int, input().split())
lst = list(map(int, input().split()))

cnt = 0
start, end = 0, 0
size, size_max = 0, 0
flag = 1

for start in range(n):
    while cnt <= k and flag:
        if lst[end] % 2:
            if cnt == k:
                break
            cnt += 1
        size += 1
        if end == n - 1:
            flag = 0
            break
        end += 1
        
    if size_max < size - cnt:
        size_max = size - cnt
        
    if lst[start] % 2:
        cnt -= 1
        
    size -= 1

print(size_max)
