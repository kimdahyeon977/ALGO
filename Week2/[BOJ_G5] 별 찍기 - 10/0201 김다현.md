def draw_stars(n):
    #print('step1',n)
    if n==1:
        return ['*']
    #print('step2',n)
    Stars=draw_stars(n//3)
    L=[]
    for star in Stars:
        L.append(star*3)
    for star in Stars:
        L.append(star+' '*(n//3)+star)
    for star in Stars:
        L.append(star*3)
    #print('step3',n,L)
    return L
N=int(input())
print('\n'.join(draw_stars(N)))
