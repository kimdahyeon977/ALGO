1.만약 종이가 모두 같은 수로 되어 있다면 이 종이를 그대로 사용한다.
2. (1)이 아닌 경우에는 종이를 같은 크기의 종이 9개로 자르고, 각각의 잘린 종이에 대해서 (1)의 과정을 반복한다.
```
```c++
#include <bits/stdc++.h>

using namespace std;
int n;
int cntm1 = 0;
int cnt0 =0;
int cnt1= 0;

vector <vector<int>> a;
void func(int x, int y, int n){
    bool flag = true;
    for(int i=x;i<x+n;i++){
        for(int j=y;j<y+n;j++){
            if(a[x][y]!=a[i][j]){
                flag = false;
            }
        }
    }
    if(flag){
        if(a[x][y]==-1){
            cntm1++;
        }else if(a[x][y]==0){
            cnt0++;
        }else{
            cnt1++;
        }
    }else{
        n=n/3;
        func(x,y,n);
        func(x+n,y,n);
        func(x+n*2,y,n);
        func(x,y+n,n);
        func(x,y+2*n,n);
        func(x+n,y+n,n);
        func(x+2*n,y+n,n);
        func(x+n,y+2*n,n);
        func(x+2*n,y+2*n,n);
    }
}
int main(){
    cin >> n;
    a.resize(n,vector<int>(n));
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            cin >> a[i][j];
        }
    }
    func(0,0,n);

    cout << cntm1 << "\n";
    cout << cnt0 << "\n";
    cout << cnt1 << "\n";

}
