```
weight를 오름차순으로 정리해야할듯?
```
```python
import sys
import heapq
input = sys.stdin.readline

n, k = map(int, input().split())
q = []
for _ in range(n):
    m, v = map(int, input().split())
    heapq.heappush(q, (-v / m , m, -v))

answer = 0
for _ in range(k):
    weight = int(input())
    while q:
        a, b, c = heapq.heappop(q)
        if b <= weight:
            answer -= c
            break
        else:
            continue
print(answer)
```
